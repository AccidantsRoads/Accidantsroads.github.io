import folium import geopandas as gpd from folium.plugins import MarkerCluster # قراءة ملفات GeoJSON accidents_gdf = gpd.read_file('Accidants.geojson') ramallah_gdf = gpd.read_file('Ramallah.geojson') roads_gdf = gpd.read_file('Roads.geojson') # إنشاء الخريطة وتحديد المركز m = folium.Map(location=[31.9, 35.2], zoom_start=12) # إضافة طبقة حوادث المرور for _, row in accidents_gdf.iterrows(): folium.Marker( location=[row['geometry'].y, row['geometry'].x], popup=f"Location: {row['geometry']}", icon=folium.Icon(color='red', icon='info-sign') ).add_to(m) # إضافة طبقة حدود رام الله folium.GeoJson( ramallah_gdf, name='Ramallah Border', style_function=lambda x: {'color': 'blue', 'weight': 2} ).add_to(m) # إضافة طبقة الطرق folium.GeoJson( roads_gdf, name='Roads', style_function=lambda x: {'color': 'gray', 'weight': 1} ).add_to(m) # إضافة أداة التحكم بالطبقات folium.LayerControl().add_to(m) # عرض الخريطة m
